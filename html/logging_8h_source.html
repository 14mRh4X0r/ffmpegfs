293  template <typename T, typename... Args>
294  static std::string format_helper(
295  const std::string &string_to_search,
296  const size_t index_to_replace,
297  T &&val,
298  Args &&...args)
299  {
300  // Match %# exactly (e.g. %12 and %123 literally)
301 
302  std::regex exp("%(<([^>]+)>)*" + std::to_string(index_to_replace) + "(?=[^0-9]|$)");
303  std::smatch res;
304  std::string string_to_update(string_to_search);
305  std::string::const_iterator searchStart(string_to_search.cbegin());
306  size_t offset = 0;
307 
308  while (std::regex_search(searchStart, string_to_search.cend(), res, exp))
309  {
310  std::ostringstream ostr;
311 
312  if (res[2].length())
313  {
314  // Found match with printf format in res[2]
315  size_t size = static_cast<size_t>(std::snprintf(nullptr, 0, res[2].str().c_str(), val)) + 1;
316  std::vector<char> buffer;
317  buffer.resize(size);
318  std::snprintf(buffer.data(), size, res[2].str().c_str(), val);
319  ostr << buffer.data();
320  }
321  else
322  {
323  // No printf format, replace literally
324  ostr << val;
325  }
326 
327  string_to_update.replace(static_cast<size_t>(res.position()) + offset, static_cast<size_t>(res[0].length()), ostr.str());
328 
329  offset += static_cast<size_t>(res.position()) + ostr.str().length();
330 
331  searchStart = res.suffix().first;
332  }
333 
334  return format_helper(
335  string_to_update,
336  index_to_replace + 1,
337  std::forward<Args>(args)...);
338  }
