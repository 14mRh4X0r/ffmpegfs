72 bool Buffer::init(bool erase_cache)
73 {
74  if (m_is_open)
75  {
76  return true;
77  }
78 
79  m_is_open = true;
80 
81  bool success = true;
82 
83  std::lock_guard<std::recursive_mutex> lck (m_mutex);
84  try
85  {
86  // Create the path to the cache file
87  char *cachefile = new_strdup(m_cachefile);
88  if (cachefile == nullptr)
89  {
90  Logging::error(m_cachefile, "Error opening cache file: Out of memory");
91  errno = ENOMEM;
92  throw false;
93  }
94 
95  struct stat sb;
96  size_t filesize;
97  void *p;
98 
99  if (mktree(dirname(cachefile), S_IRWXU | S_IRWXG | S_IROTH | S_IXOTH) && errno != EEXIST)
100  {
101  Logging::error(m_cachefile, "Error creating cache directory: %1", strerror(errno));
102  delete [] cachefile;
103  throw false;
104  }
105  errno = 0; // reset EEXIST, error can safely be ignored here
106 
107  delete [] cachefile;
108 
109  m_buffer_size = 0;
110  m_buffer = nullptr;
111  m_buffer_pos = 0;
112  m_buffer_watermark = 0;
113 
114  if (erase_cache)
115  {
116  remove_cachefile();
117  errno = 0; // ignore this error
118  }
119 
120  m_fd = ::open(m_cachefile.c_str(), O_CREAT | O_RDWR, static_cast<mode_t>(0644));
121  if (m_fd == -1)
122  {
123  Logging::error(m_cachefile, "Error opening cache file: %1", strerror(errno));
124  throw false;
125  }
126 
127  if (fstat(m_fd, &sb) == -1)
128  {
129  Logging::error(m_cachefile, "File stat failed: %1 (fd = %2)", strerror(errno), m_fd);
130  throw false;
131  }
132 
133  if (!S_ISREG(sb.st_mode))
134  {
135  Logging::error(m_cachefile, "Not a file.");
136  throw false;
137  }
138 
139  if (!sb.st_size)
140  {
141  // If empty set file size to 1 page
142  off_t pagesize = sysconf (_SC_PAGESIZE);
143 
144  if (ftruncate(m_fd, pagesize) == -1)
145  {
146  Logging::error(m_cachefile, "Error calling ftruncate() to 'stretch' the file: %1 (fd = %2)", strerror(errno), m_fd);
147  throw false;
148  }
149  filesize = static_cast<size_t>(pagesize);
150  }
151  else
152  {
153  filesize = static_cast<size_t>(sb.st_size);
154  m_buffer_pos = m_buffer_watermark = filesize;
155  }
156 
157  p = mmap(nullptr, filesize, PROT_READ | PROT_WRITE, MAP_SHARED, m_fd, 0);
158  if (p == MAP_FAILED)
159  {
160  Logging::error(m_cachefile, "File mapping failed: %1 (fd = %2)", strerror(errno), m_fd);
161  throw false;
162  }
163 
164  m_buffer_size = filesize;
165  m_buffer = static_cast<uint8_t*>(p);
166  }
167  catch (bool _success)
168  {
169  success = _success;
170 
171  if (!success)
172  {
173  m_is_open = false;
174  if (m_fd != -1)
175  {
176  ::close(m_fd);
177  m_fd = -1;
178  }
179  }
180  }
181 
182  return success;
183 }

185 bool Buffer::release(int flags /*= CLOSE_CACHE_NOOPT*/)
186 {
187  bool success = true;
188 
189  if (!m_is_open)
190  {
191  if (CACHE_CHECK_BIT(CLOSE_CACHE_DELETE, flags))
192  {
193  remove_cachefile();
194  errno = 0; // ignore this error
195  }
196 
197  return true;
198  }
199 
200  m_is_open = false;
201 
202  std::lock_guard<std::recursive_mutex> lck (m_mutex);
203 
204  // Write it now to disk
205  flush();
206 
207  void *p = m_buffer;
208  size_t size = m_buffer_size;
209  int fd = m_fd;
210 
211  m_buffer = nullptr;
212  m_buffer_size = 0;
213  m_buffer_pos = 0;
214  m_fd = -1;
215 
216  if (munmap(p, size) == -1)
217  {
218  Logging::error(m_cachefile, "File unmapping failed: %1", strerror(errno));
219  success = false;
220  }
221 
222  if (ftruncate(fd, static_cast<off_t>(m_buffer_watermark)) == -1)
223  {
224  Logging::error(m_cachefile, "Error calling ftruncate() to resize and close the file: %1 (fd = %2)", strerror(errno), fd);
225  success = false;
226  }
227 
228  ::close(fd);
229 
230  if (CACHE_CHECK_BIT(CLOSE_CACHE_DELETE, flags))
231  {
232  remove_cachefile();
233  errno = 0; // ignore this error
234  }
235 
236  return success;
237 }

360 int Buffer::seek(long offset, int whence)
361 {
362  if (!m_is_open)
363  {
364  errno = EBADF;
365  return -1;
366  }
367 
368  off_t seek_pos;
369 
370  switch (whence)
371  {
372  case SEEK_SET:
373  {
374  seek_pos = offset;
375  break;
376  }
377  case SEEK_CUR:
378  {
379  seek_pos = static_cast<off_t>(tell()) + offset;
380  break;
381  }
382  case SEEK_END:
383  {
384  seek_pos = static_cast<off_t>(size()) + offset;
385  break;
386  }
387  default:
388  {
389  errno = EINVAL;
390  return -1;
391  }
392  }
393 
394  if (seek_pos > static_cast<off_t>(size()))
395  {
396  m_buffer_pos = size(); // Cannot go beyond EOF. Set position to end, leave errno untouched.
397  return 0;
398  }
399 
400  if (seek_pos < 0) // Cannot go before head, leave position untouched, set errno.
401  {
402  errno = EINVAL;
403  return -1;
404  }
405 
406  m_buffer_pos = static_cast<uint64_t>(seek_pos);
407  return 0;
408 }
