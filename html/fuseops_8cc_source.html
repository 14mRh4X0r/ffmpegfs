106 static void init_stat(struct stat * st, size_t size, bool directory)
107 {
108  memset(st, 0, sizeof(struct stat));
109 
110  st->st_mode = DEFFILEMODE; //S_IFREG | S_IRUSR | S_IRGRP | S_IROTH;
111  if (directory)
112  {
113  st->st_mode |= S_IFDIR | S_IXUSR | S_IXGRP | S_IXOTH;
114  st->st_nlink = 2;
115  }
116  else
117  {
118  st->st_mode |= S_IFREG;
119  st->st_nlink = 1;
120  }
121 
122 #if defined __x86_64__ || !defined __USE_FILE_OFFSET64
123  st->st_size = static_cast<__off_t>(size);
124 #else
125  st->st_size = static_cast<__off64_t>(size);
126 #endif
127  st->st_blocks = (st->st_size + 512 - 1) / 512;
128 
129  // Set current user as owner
130  st->st_uid = getuid();
131  st->st_gid = getgid();
132 
133  // Use current date/time
134  st->st_atime = st->st_mtime = st->st_ctime = time(nullptr);
135 }

140 static void prepare_script()
141 {
142  std::string scriptsource;
143 
144  exepath(&scriptsource);
145  scriptsource += params.m_scriptsource;
146 
147  Logging::debug(scriptsource, "Reading virtual script source.");
148 
149  FILE *fpi = fopen(scriptsource.c_str(), "rt");
150  if (fpi == nullptr)
151  {
152  Logging::warning(scriptsource, "File open failed. Disabling script: %1", strerror(errno));
153  params.m_enablescript = false;
154  }
155  else
156  {
157  struct stat st;
158  if (fstat(fileno(fpi), &st) == -1)
159  {
160  Logging::warning(scriptsource, "File could not be accessed. Disabling script: %1", strerror(errno));
161  params.m_enablescript = false;
162  }
163  else
164  {
165  index_buffer.resize(static_cast<size_t>(st.st_size));
166 
167  if (fread(&index_buffer[0], 1, static_cast<size_t>(st.st_size), fpi) != static_cast<size_t>(st.st_size))
168  {
169  Logging::warning(scriptsource, "File could not be read. Disabling script: %1", strerror(errno));
170  params.m_enablescript = false;
171  }
172  else
173  {
174  Logging::trace(scriptsource, "Read %1 bytes of script file.", index_buffer.size());
175  }
176  }
177 
178  fclose(fpi);
179  }
180 }

285 
286  filenamemap::const_iterator it = filenames.lower_bound(path);
287  while (it != filenames.end())
288  {
289  const std::string & key = it->first;

293  struct stat stbuf;

337 LPVIRTUALFILE find_original(std::string * filepath)
338 {
339  *filepath = sanitise_name(*filepath);
340 
341  LPVIRTUALFILE virtualfile = find_file(*filepath);
342 
343  errno = 0;
344 
345  if (virtualfile != nullptr)
346  {
347  *filepath = virtualfile->m_origfile;
348  return virtualfile;
349  }
350  else
351  {
352  // Fallback to old method (required if file accessed directly)
353  std::string ext;
354  if (find_ext(&ext, *filepath) && (strcasecmp(ext, params.m_format[0].format_name()) == 0 || (params.smart_transcode() && strcasecmp(ext, params.m_format[1].format_name()) == 0)))
355  {
356  std::string dir(*filepath);
357  std::string filename(*filepath);
358  std::string tmppath;
359  struct dirent **namelist;
360  struct stat st;
361  int count;
362  int found = 0;
363 
364  remove_filename(&dir);
365  tmppath = dir;
366 
367  count = scandir(dir.c_str(), &namelist, selector, nullptr);
368  if (count == -1)
369  {
370  perror("scandir");
371  return nullptr;
372  }
373 
374  remove_path(&filename);
375  std::regex specialChars { R"([-[\]{}()*+?.,\^$|#\s])" };
376  filename = std::regex_replace(filename, specialChars, R"(\$&)");
377  replace_ext(&filename, "*");
378 
379  for (int n = 0; n < count; n++)
380  {
381  if (!found && !compare(namelist[n]->d_name, filename))
382  {
383  append_filename(&tmppath, namelist[n]->d_name);
384  found = 1;
385  }
386  free(namelist[n]);
387  }
388  free(namelist);
389 
390  tmppath = sanitise_name(tmppath);
391 
392  if (found && lstat(tmppath.c_str(), &st) == 0)
393  {
394  // File exists with this extension
395  LPVIRTUALFILE virtualfile = insert_file(VIRTUALTYPE_REGULAR, *filepath, tmppath, &st);
396  *filepath = tmppath;
397  return virtualfile;
398  }
399  else
400  {
401  // File does not exist; not an error
402  errno = 0;
403  }
404  }
405  }
406  // Source file exists with no supported extension, keep path
407  return nullptr;
408 }

453 static int ffmpegfs_readdir(const char *path, void *buf, fuse_fill_dir_t filler, off_t /*offset*/, struct fuse_file_info * /*fi*/)
454 {
455  std::string origpath;
456  DIR *dp;
457  struct dirent *de;
458 #if defined(USE_LIBBLURAY) || defined(USE_LIBDVD) || defined(USE_LIBVCD)
459  int res;
460 #endif
461 
462  Logging::trace(path, "readdir");
463 
464  translate_path(&origpath, path);
465  append_sep(&origpath);
466 
467  // Add a virtual script if enabled
468  if (params.m_enablescript)
469  {
470  std::string filename(params.m_scriptfile);
471  struct stat st;
472 
473  init_stat(&st, index_buffer.size(), false);
474 
475  if (filler(buf, filename.c_str(), &st, 0))
476  {
477  // break;
478  }
479 
480  insert_file(VIRTUALTYPE_SCRIPT, origpath + filename, &st);
481  }
482 
483 #ifdef USE_LIBVCD
484  res = check_vcd(origpath, buf, filler);
485  if (res != 0)
486  {
487  // Found VCD or error reading VCD
488  return (res >= 0 ? 0 : res);
489  }
490 #endif // USE_LIBVCD
491 #ifdef USE_LIBDVD
492  res = check_dvd(origpath, buf, filler);
493  if (res != 0)
494  {
495  // Found DVD or error reading DVD
496  return (res >= 0 ? 0 : res);
497  }
498 #endif // USE_LIBDVD
499 #ifdef USE_LIBBLURAY
500  res = check_bluray(origpath, buf, filler);
501  if (res != 0)
502  {
503  // Found Bluray or error reading Bluray
504  return (res >= 0 ? 0 : res);
505  }
506 #endif // USE_LIBBLURAY
507 
508  dp = opendir(origpath.c_str());
509  if (dp != nullptr)
510  {
511  try
512  {
513  while ((de = readdir(dp)) != nullptr)
514  {
515  std::string filename(de->d_name);
516  std::string origfile;
517  struct stat st;
518 
519  origfile = origpath + filename;
520 
521  if (lstat(origfile.c_str(), &st) == -1)
522  {
523  throw false;
524  }
525 
526  if (S_ISREG(st.st_mode) || S_ISLNK(st.st_mode))
527  {
528  FFmpegfs_Format *current_format = nullptr;
529 
530  if (transcoded_name(&filename, &current_format))
531  {
532  insert_file(VIRTUALTYPE_REGULAR, origpath + filename, origfile, &st);
533  }
534  }
535 
536  if (filler(buf, filename.c_str(), &st, 0))
537  {
538  break;
539  }
540  }
541 
542  errno = 0; // Just to make sure - reset any error
543  }
544  catch (bool)
545  {
546 
547  }
548 
549  closedir(dp);
550  }
551 
552  return -errno;
553 }

561 static int ffmpegfs_getattr(const char *path, struct stat *stbuf)
562 {
563  std::string origpath;
564 
565  Logging::trace(path, "getattr");
566 
567  translate_path(&origpath, path);
568 
569  if (lstat(origpath.c_str(), stbuf) == 0)
570  {
571  // pass-through for regular files
572  errno = 0;
573  return 0;
574  }
575  else
576  {
577  // Not really an error.
578  errno = 0;
579  }
580 
581  // This is a virtual file
582  LPVIRTUALFILE virtualfile = find_original(&origpath);
583  VIRTUALTYPE type = (virtualfile != nullptr) ? virtualfile->m_type : VIRTUALTYPE_REGULAR;
584 
585  bool no_lstat = false;
586 
587  switch (type)
588  {
589  case VIRTUALTYPE_SCRIPT:
590  {
591  // Use stored status
592  mempcpy(stbuf, &virtualfile->m_st, sizeof(struct stat));
593  errno = 0;
594  break;
595  }
596 #ifdef USE_LIBVCD
597  case VIRTUALTYPE_VCD:
598 #endif // USE_LIBVCD
599 #ifdef USE_LIBDVD
600  case VIRTUALTYPE_DVD:
601 #endif // USE_LIBDVD
602 #ifdef USE_LIBBLURAY
603  case VIRTUALTYPE_BLURAY:
604 #endif // USE_LIBBLURAY
605  {
606  // Use stored status
607  mempcpy(stbuf, &virtualfile->m_st, sizeof(struct stat));
608  no_lstat = true;
609  [[clang::fallthrough]];
610  }
611  case VIRTUALTYPE_REGULAR:
612  {
613  if (!no_lstat)
614  {
615  if (lstat(origpath.c_str(), stbuf) == -1)
616  {
617  int error = -errno;
618 #if defined(USE_LIBBLURAY) || defined(USE_LIBDVD) || defined(USE_LIBVCD)
619  int res = 0;
620 
621  std::string _origpath(origpath);
622 
623  remove_filename(&_origpath);
624 
625  virtualfile = find_original(&origpath);
626 
627  if (virtualfile == nullptr)
628  {
629 #ifdef USE_LIBVCD
630  if (res <= 0)
631  {
632  // Returns -errno or number or titles on VCD
633  res = check_vcd(_origpath);
634  }
635 #endif // USE_LIBVCD
636 #ifdef USE_LIBDVD
637  if (res <= 0)
638  {
639  // Returns -errno or number or titles on DVD
640  res = check_dvd(_origpath);
641  }
642 #endif // USE_LIBDVD
643 #ifdef USE_LIBBLURAY
644  if (res <= 0)
645  {
646  // Returns -errno or number or titles on Bluray
647  res = check_bluray(_origpath);
648  }
649 #endif // USE_LIBBLURAY
650  if (res <= 0)
651  {
652  // No Bluray/DVD/VCD found or error reading disk
653  return (!res ? error : res);
654  }
655  }
656 
657  virtualfile = find_original(&origpath);
658 
659  if (virtualfile == nullptr)
660  {
661  // Not a DVD/VCD/Bluray file
662  return -ENOENT;
663  }
664 
665  mempcpy(stbuf, &virtualfile->m_st, sizeof(struct stat));
666 #else
667  return error;
668 #endif
669  }
670  }
671 
672  // Get size for resulting output file from regular file, otherwise it's a symbolic link.
673  if (S_ISREG(stbuf->st_mode))
674  {
675  assert(virtualfile->m_origfile == origpath);
676 
677  if (!transcoder_cached_filesize(virtualfile, stbuf))
678  {
679  Cache_Entry* cache_entry = transcoder_new(virtualfile, false);
680  if (cache_entry == nullptr)
681  {
682  return -errno;
683  }
684 
685 #if defined __x86_64__ || !defined __USE_FILE_OFFSET64
686  stbuf->st_size = static_cast<__off_t>(transcoder_get_size(cache_entry));
687 #else
688  stbuf->st_size = static_cast<__off64_t>(transcoder_get_size(cache_entry));
689 #endif
690  stbuf->st_blocks = (stbuf->st_size + 512 - 1) / 512;
691 
692  transcoder_delete(cache_entry);
693  }
694  }
695 
696  errno = 0; // Just to make sure - reset any error
697  break;
698  }
699  case VIRTUALTYPE_PASSTHROUGH: // We should never come here but this shuts up a warning
700  case VIRTUALTYPE_BUFFER:
701  {
702  assert(false);
703  break;
704  }
705  }
706 
707  return 0;
708 }
